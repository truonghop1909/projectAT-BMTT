#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Keypad_I2C.h>
#include <Keypad.h>
#include <SPI.h>
#include <MFRC522.h>
#include <ESP32Servo.h>
#include <Adafruit_NeoPixel.h>

// ⚡ Fix conflict Keypad_I2C và PCF8574
#undef PCF8574
#include "PCF8574.h"

#include "BluetoothSerial.h"
#if !defined(CONFIG_BT_ENABLED) || !defined(CONFIG_BLUEDROID_ENABLED)
#error Board hien tai khong ho tro Bluetooth Classic. Hay chon ESP32 Dev Module.
#endif
BluetoothSerial SerialBT;

// ================== LCD I2C ==================
LiquidCrystal_I2C lcd01(0x27, 16, 2);  // LCD địa chỉ 27

// ================== LCD I2C BỔ SUNG ==================
LiquidCrystal_I2C lcd02(0x26, 16, 2);  // LCD địa chỉ 26 

// ================== Keypad qua PCF8574 ==================
#define KEYPAD_ADDR 0x20
const byte ROWS = 4, COLS = 4;
char keys[ROWS][COLS] = {
  { '1', '2', '3', 'A' },
  { '4', '5', '6', 'B' },
  { '7', '8', '9', 'C' },
  { '*', '0', '#', 'D' }
};
byte rowPins[ROWS] = { 0, 1, 2, 3 };
byte colPins[COLS] = { 4, 5, 6, 7 };
Keypad_I2C keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS, KEYPAD_ADDR, 1, &Wire);

// ================== RFID RC522 ==================
#define SS_PIN 5
#define RST_PIN 4
MFRC522 mfrc522(SS_PIN, RST_PIN);

// ================== Servo ==================
Servo doorServo;
#define SERVO_PIN 25

// ================== Cảm biến & Relay ==================
#define GAS_SENSOR 34    // MQ-2 (AO analog)
#define FLAME_SENSOR 35  // Flame sensor A0
#define RELAY_FAN 27     // quạt
#define RELAY_PUMP 26    // bơm
#define BUZZER_PIN 14    // buzzer
#define LED_ALARM 12     // LED đỏ cảnh báo 🔴
#define LDR_SENSOR 32    // cảm biến ánh sáng (DO -> GPIO36 ADC1_CH0)


// ================== Hệ thống đèn ==================
#define LIGHT_OUT 17  // đèn ngoài (relay)
#define LIGHT_LIV 15  // đèn khách (relay)
#define LIGHT_BED 2   // đèn ngủ (NeoPixel)

#define NUMPIXELS 12
Adafruit_NeoPixel pixels(NUMPIXELS, LIGHT_BED, NEO_GRB + NEO_KHZ800);

// ================== Giàn phơi ==================
#define RAIN_SENSOR 16    // cảm biến mưa DO
Servo dryServo;           // servo giàn phơi
#define DRY_SERVO_PIN 13  // chân servo phơi

bool dryingOut = false;  // trạng thái giàn phơi (false = thu vào, true = trải ra)


// ================== PCF8574 Buttons ==================
PCF8574 pcf(0x21);
#define BTN_DOOR 0
#define BTN_FAN 1
#define BTN_PUMP 2
#define BTN_LIGHT 3
#define BTN_L_OUT 4
#define BTN_L_LIV 5
#define BTN_L_BED 6
#define BTN_FIRE 7


// ================== Biến toàn cục ==================
String passwordSet = "1234";
String inputPass = "";
byte validUID[4] = { 0x3A, 0x4D, 0x32, 0x16 };
bool fireAlertSent = false;
bool gasAlertSent = false;
bool fanState = false;
bool pumpState = false;
bool alarmMuted = false;

bool lightOut = false, lightLiv = false, lightBed = false;
bool fireDoorOpened = false;

bool changePassMode = false;  // đang trong chế độ đổi mật khẩu
int changeStep = 0;           // bước đổi mật khẩu (0 = chưa, 1 = nhập cũ, 2 = nhập mới)
String newPassword = "";      // để lưu mật khẩu mới

bool lightOutAuto = true;  // true = auto, false = manual OFF

int bedColorIndex = 0;  // lưu trạng thái màu hiện tại

// ================== HÀM MỞ / ĐÓNG CỬA MƯỢT ==================

// Mở cửa mượt
void openDoorSmooth() {
  lcd01.clear();
  lcd01.print("Mo cua...");
  moveServoSmooth(doorServo, 0, 90, 12);  // bước 1°, delay 12ms
}

// Đóng cửa mượt
void closeDoorSmooth() {
  lcd01.clear();
  lcd01.print("Dong cua...");
  moveServoSmooth(doorServo, 90, 0, 12);  // bước 1°, delay 12ms
  lcd01.clear();
  lcd01.print("Nhap mat khau:");
}

// Mở rồi đóng mượt (tự động đóng sau 3 giây)
void openAndCloseDoorSmooth() {
  openDoorSmooth();
  delay(3000);
  closeDoorSmooth();
}




// ================== NeoPixel Helper ==================
void neoPixelOn() {
  for (int i = 0; i < NUMPIXELS; i++) {
    pixels.setPixelColor(i, pixels.Color(255, 255, 255));  // sáng trắng
  }
  pixels.show();
}
void neoPixelOff() {
  for (int i = 0; i < NUMPIXELS; i++) {
    pixels.setPixelColor(i, 0);  // tắt
  }
  pixels.show();
}
// ================== SMOOTH SERVO PHOI ==================
void moveServoSmooth(Servo &servo, int startAngle, int endAngle, int stepDelay) {
  if (startAngle < endAngle) {
    for (int pos = startAngle; pos <= endAngle; pos++) {
      servo.write(pos);
      delay(stepDelay);  // delay giữa các bước
    }
  } else {
    for (int pos = startAngle; pos >= endAngle; pos--) {
      servo.write(pos);
      delay(stepDelay);
    }
  }
}


// ================== BUZZERBEEP ==================

void buzzerBeep(int times, int duration, int pause) {
  for (int i = 0; i < times; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    delay(duration);
    digitalWrite(BUZZER_PIN, LOW);
    if (i < times - 1) delay(pause);  // nghỉ giữa các tiếng beep
  }
}

// ================== NEOPIXEL12LED ==================

void setBedColor(int index) {
  if (index == 0) {
    neoPixelOff();  // tắt hẳn
  } else {
    uint32_t color;
    switch (index) {
      case 1: color = pixels.Color(255, 0, 0); break;      // đỏ
      case 2: color = pixels.Color(0, 255, 0); break;      // xanh lá
      case 3: color = pixels.Color(0, 0, 255); break;      // xanh dương
      case 4: color = pixels.Color(255, 255, 0); break;    // vàng
      case 5: color = pixels.Color(255, 0, 255); break;    // tím
      case 6: color = pixels.Color(0, 255, 255); break;    // cyan
      case 7: color = pixels.Color(255, 255, 255); break;  // trắng
      default: color = 0; break;
    }

    if (index > 0 && index <= 7) {
      for (int i = 0; i < NUMPIXELS; i++) {
        pixels.setPixelColor(i, color);
      }
      pixels.show();
    }
  }
}



// ================== Setup ==================
void setup() {
  Serial.begin(115200);
  Wire.begin(21, 22);

  // LCD
  lcd01.init();
  lcd02.init();
  lcd01.backlight();
  lcd02.backlight();
  lcd01.print("He thong khoi");
  lcd02.print("He thong khoi");
  lcd01.setCursor(0, 1);
  lcd02.setCursor(0, 1);
  lcd01.print("dong...");
  lcd02.print("dong...");
  delay(2000);
  lcd01.clear();
  lcd02.clear();
  lcd01.print("Nhap mat khau:");
  lcd02.print("Active:");

  // Keypad
  keypad.begin();

  // RFID
  SPI.begin(18, 19, 23, 5);
  mfrc522.PCD_Init();

  // Servo
  doorServo.setPeriodHertz(50);
  doorServo.attach(SERVO_PIN, 500, 2400);
  doorServo.write(0);

  // IO
  pinMode(FLAME_SENSOR, INPUT);
  pinMode(GAS_SENSOR, INPUT);
  pinMode(RELAY_FAN, OUTPUT);
  pinMode(RELAY_PUMP, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(LED_ALARM, OUTPUT);
  pinMode(LIGHT_LIV, OUTPUT);
  digitalWrite(RELAY_FAN, HIGH);
  digitalWrite(RELAY_PUMP, HIGH);
  digitalWrite(BUZZER_PIN, LOW);
  digitalWrite(LED_ALARM, LOW);
  pinMode(LDR_SENSOR, INPUT);
  pinMode(LIGHT_OUT, OUTPUT);
  digitalWrite(LIGHT_OUT, LOW);
  digitalWrite(LIGHT_LIV, LOW);

  // NeoPixel
  pixels.begin();
  neoPixelOff();

  // PCF8574
  if (!pcf.begin()) {
    Serial.println("❌ Không tìm thấy PCF8574!");
    while (1)
      ;
  }
  for (int i = 0; i < 7; i++) pcf.write(i, HIGH);  // input pull-up

  // Giàn phơi
  pinMode(RAIN_SENSOR, INPUT);
  dryServo.setPeriodHertz(50);
  dryServo.attach(DRY_SERVO_PIN, 500, 2400);
  dryServo.write(0);  // mặc định thu vào


  // Khởi tạo Bluetooth
  if (!SerialBT.begin("ESP32_SMARTHOME")) {  // Tên thiết bị Bluetooth
    Serial.println("❌ Bluetooth không khởi động được!");
  } else {
    Serial.println("✅ Bluetooth sẵn sàng, tên: ESP32_SMARTHOME");
  }

  delay(3000);
  Serial.println("Hệ thống sẵn sàng!");
}
// ================== Loop ==================
void loop() {

  // -------- PHẦN 1: BÁO CHÁY / GAS --------
  int flameValue = analogRead(FLAME_SENSOR);
  int gasValue = analogRead(GAS_SENSOR);
  int flameThreshold = 500;
  Serial.print("🔥 Gia tri cam bien lua: ");
  Serial.println(flameValue);

  static unsigned long dangerEndTime = 0;
  static bool dangerActive = false;

  if (!alarmMuted && (flameValue < flameThreshold || gasValue > 2000)) {
    // 🚨 Có cháy hoặc gas
    dangerActive = true;
    dangerEndTime = millis() + 5000;  // giữ thêm 5s sau khi hết tín hiệu

    digitalWrite(BUZZER_PIN, HIGH);  // còi hú liên tục
    digitalWrite(LED_ALARM, HIGH);   // đèn đỏ sáng

    // Quạt, bơm
    if (flameValue < flameThreshold) {  // cháy
      digitalWrite(RELAY_FAN, LOW);
      digitalWrite(RELAY_PUMP, LOW);

      // Ngắt toàn bộ đèn
      digitalWrite(LIGHT_OUT, HIGH);  // OFF đèn ngoài
      digitalWrite(LIGHT_LIV, LOW);   // OFF đèn khách
      neoPixelOff();                  // OFF đèn ngủ
      lcd01.clear();
      lcd01.print("🔥 LUA! CANH BAO");

      if (!fireDoorOpened) {
        openDoorSmooth();
        fireDoorOpened = true;
      }

      if (!fireAlertSent) {
        fireAlertSent = true;
      }
    } else if (gasValue > 2000) {  // gas
      digitalWrite(RELAY_FAN, LOW);
      digitalWrite(RELAY_PUMP, HIGH);
      // Ngắt toàn bộ đèn
      digitalWrite(LIGHT_OUT, HIGH);  // OFF đèn ngoài
      digitalWrite(LIGHT_LIV, LOW);   // OFF đèn khách
      neoPixelOff();                  // OFF đèn ngủ
      lcd01.clear();
      lcd01.print("⚠️ GAS! CANH BAO");

      if (!fireDoorOpened) {
        openDoorSmooth();
        fireDoorOpened = true;
      }

      if (!gasAlertSent) {
        gasAlertSent = true;
      }
    }

  } else if (!alarmMuted) {
    // ❓ Không còn tín hiệu cháy/gas
    if (dangerActive) {
      if (millis() < dangerEndTime) {
        // ⏳ Trong 5s giữ còi + đèn sáng, cửa vẫn mở
        digitalWrite(BUZZER_PIN, HIGH);
        digitalWrite(LED_ALARM, HIGH);

      } else {
        // ✅ Sau 5s -> tắt còi, đèn, đóng cửa
        dangerActive = false;

        digitalWrite(BUZZER_PIN, LOW);
        digitalWrite(LED_ALARM, LOW);

        if (fireDoorOpened) {
          closeDoorSmooth();
          fireDoorOpened = false;
        }

        digitalWrite(RELAY_FAN, fanState ? LOW : HIGH);
        digitalWrite(RELAY_PUMP, pumpState ? LOW : HIGH);

        lcd01.setCursor(0, 0);
        lcd01.print("Nhap mat khau:   ");

        fireAlertSent = false;
        gasAlertSent = false;
      }
    }
  }


  // -------- PHẦN 2: NÚT PCF8574 --------

  // ======= Nút 0: CỬA =======
  if (pcf.read(BTN_DOOR) == LOW) {
    buzzerBeep(1, 100, 0);  // 🔔 kêu 1 tít khi nhấn
    if (fireDoorOpened) {
      closeDoorSmooth();
      fireDoorOpened = false;
      lcd02.clear();
      lcd02.print("Cua: DONG");
    } else {
      openDoorSmooth();
      fireDoorOpened = true;
      lcd02.clear();
      lcd02.print("Cua: MO");
    }
    delay(300);
  }

  // ======= Nút 1: QUẠT =======
  if (pcf.read(BTN_FAN) == LOW) {
    buzzerBeep(1, 100, 0);
    fanState = !fanState;
    digitalWrite(RELAY_FAN, fanState ? LOW : HIGH);
    Serial.println(fanState ? "Quạt ON" : "Quạt OFF");
    lcd02.clear();
    lcd02.print(fanState ? "Quat: ON" : "Quat: OFF");
    delay(300);
  }

  // ======= Nút 2: BƠM =======
  if (pcf.read(BTN_PUMP) == LOW) {
    buzzerBeep(1, 100, 0);
    pumpState = !pumpState;
    digitalWrite(RELAY_PUMP, pumpState ? LOW : HIGH);
    Serial.println(pumpState ? "Bơm ON" : "Bơm OFF");
    lcd02.clear();
    lcd02.print(pumpState ? "Bom: ON" : "Bom: OFF");
    delay(300);
  }

  // ======= Nút 3: BẬT / TẮT TOÀN BỘ ĐÈN =======
  static bool allLightsOn = false;
  if (pcf.read(BTN_LIGHT) == LOW) {
    buzzerBeep(1, 100, 0);
    allLightsOn = !allLightsOn;

    if (allLightsOn) {
      digitalWrite(LIGHT_OUT, LOW);   // Bật đèn ngoài (LOW = ON)
      digitalWrite(LIGHT_LIV, HIGH);  // Bật đèn khách
      setBedColor(7);                 // Đèn ngủ trắng
      lcd02.clear();
      lcd02.print("Tat ca den: ON");
      Serial.println("💡 Bật tất cả đèn");
    } else {
      digitalWrite(LIGHT_OUT, HIGH);  // Tắt đèn ngoài (HIGH = OFF)
      digitalWrite(LIGHT_LIV, LOW);   // Tắt đèn khách
      setBedColor(0);                 // Tắt đèn ngủ
      lcd02.clear();
      lcd02.print("Tat ca den: OFF");
      Serial.println("💡 Tắt tất cả đèn");
    }
    delay(300);
  }


  // ======= Nút 4: ĐÈN NGOÀI =======
  if (pcf.read(BTN_L_OUT) == LOW) {
    buzzerBeep(1, 100, 0);
    lightOutAuto = !lightOutAuto;
    if (!lightOutAuto) {
      digitalWrite(LIGHT_OUT, LOW);  // ép bật tay
      Serial.println("Đèn ngoài: MANUAL ON");
      lcd02.clear();
      lcd02.print("Den ngoai: ON");
    } else {
      digitalWrite(LIGHT_OUT, HIGH);  // tắt auto
      Serial.println("Đèn ngoài: AUTO OFF");
      lcd02.clear();
      lcd02.print("Den ngoai: AUTO");
    }
    delay(300);
  }

  // ======= Nút 5: ĐÈN KHÁCH =======
  if (pcf.read(BTN_L_LIV) == LOW) {
    buzzerBeep(1, 100, 0);
    lightLiv = !lightLiv;
    digitalWrite(LIGHT_LIV, lightLiv ? HIGH : LOW);
    Serial.println(lightLiv ? "Đèn khách ON" : "Đèn khách OFF");
    lcd02.clear();
    lcd02.print(lightLiv ? "Den khach: ON" : "Den khach: OFF");
    delay(300);
  }

  // ======= Nút 6: ĐÈN NGỦ (CHUYỂN MÀU) =======
  if (pcf.read(BTN_L_BED) == LOW) {
    buzzerBeep(1, 100, 0);
    bedColorIndex++;
    if (bedColorIndex > 7) bedColorIndex = 0;
    setBedColor(bedColorIndex);

    lcd02.clear();
    if (bedColorIndex == 0) {
      lcd02.print("Den ngu: OFF");
      Serial.println("Đèn ngủ OFF");
    } else {
      lcd02.print("Den ngu: Mau ");
      lcd02.print(bedColorIndex);
      Serial.print("Đèn ngủ màu ");
      Serial.println(bedColorIndex);
    }
    delay(300);
  }

  // ======= Nút 7: CHẾ ĐỘ CẢNH BÁO KHẨN =======
  static bool emergencyMode = false;
  if (pcf.read(BTN_FIRE) == LOW) {
    buzzerBeep(1, 100, 0);
    emergencyMode = !emergencyMode;

    if (emergencyMode) {
      // 🚨 Bật cảnh báo
      digitalWrite(BUZZER_PIN, HIGH);
      digitalWrite(LED_ALARM, HIGH);
      digitalWrite(RELAY_FAN, LOW);
      digitalWrite(RELAY_PUMP, LOW);
      if (!fireDoorOpened) {
        openDoorSmooth();
        fireDoorOpened = true;
      }

      // Ngắt toàn bộ đèn
      digitalWrite(LIGHT_OUT, HIGH);
      digitalWrite(LIGHT_LIV, LOW);
      setBedColor(0);

      lcd02.clear();
      lcd02.print("CANH BAO: ON");
      Serial.println("🚨 CẢNH BÁO KÍCH HOẠT");
    } else {
      // 🔕 Tắt cảnh báo
      digitalWrite(BUZZER_PIN, LOW);
      digitalWrite(LED_ALARM, LOW);
      digitalWrite(RELAY_FAN, HIGH);
      digitalWrite(RELAY_PUMP, HIGH);
      if (fireDoorOpened) {
        closeDoorSmooth();
        fireDoorOpened = false;
      }

      lcd02.clear();
      lcd02.print("CANH BAO: OFF");
      Serial.println("✅ CẢNH BÁO TẮT");
      lcd01.clear();
      lcd01.print("Nhap mat khau:");
    }
    delay(300);
  }

  // -------- PHẦN 4: HỆ THỐNG CỬA (RFID + Keypad) --------
  if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
    bool valid = true;
    if (mfrc522.uid.size == 4) {
      for (byte i = 0; i < 4; i++) {
        if (mfrc522.uid.uidByte[i] != validUID[i]) {
          valid = false;
          break;
        }
      }
    } else valid = false;

    if (valid) {
      buzzerBeep(2, 100, 100);  // ✅ đúng -> 2 tít ngắn
      openAndCloseDoorSmooth();
    } else {
      buzzerBeep(1, 1000, 0);  // ❌ sai -> 1 tiếng dài
      lcd01.clear();
      lcd01.print("The sai!");
      delay(2000);
      lcd01.clear();
      lcd01.print("Nhap mat khau:");
    }
    mfrc522.PICC_HaltA();
    mfrc522.PCD_StopCrypto1();
  }

  // Keypad
  char key = keypad.getKey();
  if (key) {
    buzzerBeep(1, 100, 0);  // 🔘 mỗi lần nhấn phím -> kêu 1 tít ngắn

    if (!changePassMode) {
      // ======== CHẾ ĐỘ BÌNH THƯỜNG ========
      if (key == '#') {
        if (inputPass == passwordSet) {
          buzzerBeep(2, 100, 100);  // ✅ đúng -> 2 tít
          openAndCloseDoorSmooth();
        } else {
          buzzerBeep(1, 1000, 0);  // ❌ sai -> 1 dài
          lcd01.clear();
          lcd01.print("Sai mat khau!");
          delay(2000);
          lcd01.clear();
          lcd01.print("Nhap mat khau:");
        }
        inputPass = "";
      } else if (key == '*') {
        inputPass = "";
        lcd01.clear();
        lcd01.print("Nhap mat khau:");
      } else if (key == 'A') {
        // Bắt đầu đổi mật khẩu
        changePassMode = true;
        changeStep = 1;
        inputPass = "";
        lcd01.clear();
        lcd01.print("Nhap MK cu:");
      } else {
        inputPass += key;
        lcd01.setCursor(0, 1);
        lcd01.print(inputPass);
      }
    } else {
      // ======== CHẾ ĐỘ ĐỔI MẬT KHẨU ========
      if (key == '#') {
        if (changeStep == 1) {
          // kiểm tra mật khẩu cũ
          if (inputPass == passwordSet) {
            lcd01.clear();
            lcd01.print("Nhap MK moi:");
            changeStep = 2;
            inputPass = "";
          } else {
            lcd01.clear();
            lcd01.print("MK cu sai!");
            buzzerBeep(1, 1000, 0);
            delay(2000);
            lcd01.clear();
            lcd01.print("Nhap mat khau:");
            changePassMode = false;
            changeStep = 0;
            inputPass = "";
          }
        } else if (changeStep == 2) {
          // lưu mật khẩu mới
          passwordSet = inputPass;
          lcd01.clear();
          lcd01.print("MK da doi!");
          buzzerBeep(2, 200, 200);
          delay(2000);
          lcd01.clear();
          lcd01.print("Nhap mat khau:");
          changePassMode = false;
          changeStep = 0;
          inputPass = "";
        }
      } else if (key == '*') {
        inputPass = "";
        lcd01.clear();
        if (changeStep == 1) lcd01.print("Nhap MK cu:");
        else if (changeStep == 2) lcd01.print("Nhap MK moi:");
      } else {
        inputPass += key;
        lcd01.setCursor(0, 1);
        lcd01.print(inputPass);
      }
    }
  }

  // -------- PHẦN 5: GIÀN PHƠI TỰ ĐỘNG (ngược lại, quay mượt) --------
  int rainState = digitalRead(RAIN_SENSOR);
  if (rainState == LOW) {  // có mưa
    if (!dryingOut) {
      Serial.println("🌧 Có mưa -> Thu giàn phơi");
      moveServoSmooth(dryServo, 0, 90, 20);  // từ 0 -> 90 độ, mỗi bước 20ms
      dryingOut = true;
      lcd02.clear();
      lcd02.print("Mua: Thu Gian Phoi");
    }
  } else {  // không mưa
    if (dryingOut) {
      Serial.println("☀️ Trời khô -> Đưa giàn phơi ra");
      moveServoSmooth(dryServo, 90, 0, 20);  // từ 90 -> 0 độ
      dryingOut = false;
      lcd02.clear();
      lcd02.print("Nang Phoi Do");
    }
  }

  // -------- AUTO ĐÈN NGOÀI --------
  if (lightOutAuto) {
    int lightState = digitalRead(LDR_SENSOR);
    if (lightState == LOW) {          // trời tối
      digitalWrite(LIGHT_OUT, LOW);   // bật relay -> đèn ON
    } else {                          // trời sáng
      digitalWrite(LIGHT_OUT, HIGH);  // tắt relay -> đèn OFF
    }
  }

  // -------- BLUETOOTH COMMAND HANDLER --------
  if (SerialBT.available()) {
    String cmd = SerialBT.readStringUntil('\n');
    cmd.trim();
    Serial.print("📢 CMD: ");
    Serial.println(cmd);

    // ========== 1. RGB Picker ==========
    if (cmd.length() == 9 && isDigit(cmd[0])) {
      int r = cmd.substring(0, 3).toInt();
      int g = cmd.substring(3, 6).toInt();
      int b = cmd.substring(6, 9).toInt();

      for (int i = 0; i < NUMPIXELS; i++) {
        pixels.setPixelColor(i, pixels.Color(r, g, b));
      }
      pixels.show();

      SerialBT.println("✅ LED đổi màu!");
      lcd02.clear();
      lcd02.print("Den ngu: RGB");
    }

    // ========== 2. Door ==========
    else if (cmd.equalsIgnoreCase("OPEN DOOR")) {
      openDoorSmooth();
      fireDoorOpened = true;
      SerialBT.println("🚪 Door opened");
      lcd02.clear();
      lcd02.print("Cua: MO");
    } else if (cmd.equalsIgnoreCase("CLOSE DOOR")) {
      closeDoorSmooth();
      fireDoorOpened = false;
      SerialBT.println("🚪 Door closed");
      lcd02.clear();
      lcd02.print("Cua: DONG");
    }

    // ========== 3. Fan ==========
    else if (cmd.equalsIgnoreCase("FAN ON")) {
      fanState = true;
      digitalWrite(RELAY_FAN, LOW);
      SerialBT.println("🌀 Fan ON");
      lcd02.clear();
      lcd02.print("Quat: ON");
    } else if (cmd.equalsIgnoreCase("FAN OFF")) {
      fanState = false;
      digitalWrite(RELAY_FAN, HIGH);
      SerialBT.println("🌀 Fan OFF");
      lcd02.clear();
      lcd02.print("Quat: OFF");
    }

    // ========== 4. Pump ==========
    else if (cmd.equalsIgnoreCase("PUMP ON")) {
      pumpState = true;
      digitalWrite(RELAY_PUMP, LOW);
      SerialBT.println("💧 Pump ON");
      lcd02.clear();
      lcd02.print("Bom: ON");
    } else if (cmd.equalsIgnoreCase("PUMP OFF")) {
      pumpState = false;
      digitalWrite(RELAY_PUMP, HIGH);
      SerialBT.println("💧 Pump OFF");
      lcd02.clear();
      lcd02.print("Bom: OFF");
    }

    // ========== 5. Lights ==========
    else if (cmd.equalsIgnoreCase("ALL LIGHTS ON")) {
      digitalWrite(LIGHT_OUT, LOW);
      digitalWrite(LIGHT_LIV, HIGH);
      setBedColor(7);
      SerialBT.println("💡 All lights ON");
      lcd02.clear();
      lcd02.print("Tat ca den: ON");
    } else if (cmd.equalsIgnoreCase("ALL LIGHTS OFF")) {
      digitalWrite(LIGHT_OUT, HIGH);
      digitalWrite(LIGHT_LIV, LOW);
      setBedColor(0);
      SerialBT.println("💡 All lights OFF");
      lcd02.clear();
      lcd02.print("Tat ca den: OFF");
    } else if (cmd.equalsIgnoreCase("OUTSIDE LIGHT ON")) {
      digitalWrite(LIGHT_OUT, LOW);
      SerialBT.println("💡 Outside light ON");
      lcd02.clear();
      lcd02.print("Den ngoai: ON");
    } else if (cmd.equalsIgnoreCase("OUTSIDE LIGHT OFF")) {
      digitalWrite(LIGHT_OUT, HIGH);
      SerialBT.println("💡 Outside light OFF");
      lcd02.clear();
      lcd02.print("Den ngoai: OFF");
    } else if (cmd.equalsIgnoreCase("LIVING LIGHT ON")) {
      digitalWrite(LIGHT_LIV, HIGH);
      SerialBT.println("💡 Living room light ON");
      lcd02.clear();
      lcd02.print("Den khach: ON");
    } else if (cmd.equalsIgnoreCase("LIVING LIGHT OFF")) {
      digitalWrite(LIGHT_LIV, LOW);
      SerialBT.println("💡 Living room light OFF");
      lcd02.clear();
      lcd02.print("Den khach: OFF");
    }

    // ========== 6. Bedroom NeoPixel ==========
    else if (cmd.equalsIgnoreCase("BEDROOM OFF")) {
      setBedColor(0);
      SerialBT.println("⚫ Bedroom OFF");
      lcd02.clear();
      lcd02.print("Den ngu: OFF");
    } else if (cmd.equalsIgnoreCase("BEDROOM RED")) {
      setBedColor(1);
      SerialBT.println("🟥 Bedroom RED");
      lcd02.clear();
      lcd02.print("Den ngu: Do");
    } else if (cmd.equalsIgnoreCase("BEDROOM GREEN")) {
      setBedColor(2);
      SerialBT.println("🟩 Bedroom GREEN");
      lcd02.clear();
      lcd02.print("Den ngu: Xanh");
    } else if (cmd.equalsIgnoreCase("BEDROOM BLUE")) {
      setBedColor(3);
      SerialBT.println("🟦 Bedroom BLUE");
      lcd02.clear();
      lcd02.print("Den ngu: Xanh Duong");
    } else if (cmd.equalsIgnoreCase("BEDROOM WHITE")) {
      setBedColor(7);
      SerialBT.println("⬜ Bedroom WHITE");
      lcd02.clear();
      lcd02.print("Den ngu: Trang");
    }

    // ========== 7. Emergency Mode ==========
    else if (cmd.equalsIgnoreCase("ALERT ON")) {
      digitalWrite(BUZZER_PIN, HIGH);
      digitalWrite(LED_ALARM, HIGH);
      digitalWrite(RELAY_FAN, LOW);
      digitalWrite(RELAY_PUMP, LOW);
      if (!fireDoorOpened) {
        openDoorSmooth();
        fireDoorOpened = true;
      }
      digitalWrite(LIGHT_OUT, HIGH);
      digitalWrite(LIGHT_LIV, LOW);
      setBedColor(0);
      SerialBT.println("🚨 Emergency Mode ON");
      lcd02.clear();
      lcd02.print("CANH BAO: ON");
    } else if (cmd.equalsIgnoreCase("ALERT OFF")) {
      digitalWrite(BUZZER_PIN, LOW);
      digitalWrite(LED_ALARM, LOW);
      digitalWrite(RELAY_FAN, HIGH);
      digitalWrite(RELAY_PUMP, HIGH);
      if (fireDoorOpened) {
        closeDoorSmooth();
        fireDoorOpened = false;
      }
      SerialBT.println("✅ Emergency Mode OFF");
      lcd02.clear();
      lcd02.print("CANH BAO: OFF");
    }

    // ========== 8. Unknown ==========
    else {
      SerialBT.println("❓ Unknown command");
    }
  }
}
